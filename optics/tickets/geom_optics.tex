\subsection{–ü—Ä–∏–Ω—Ü–∏–ø –§–µ—Ä–º–∞}

–í—Å–ø–æ–º–Ω–∏–º —É—Ä–∞–≤–Ω–µ–Ω–∏—è –ú–∞–∫—Å–≤–µ–ª–ª–∞
\begin{equation*}
    \left\{\begin{aligned}
            \div \vc{D} &= 4 \pi \rho \\
            \div \vc{B} &= 0 \\
            \rot \vc{E} &= - \frac{1}{c}\frac{\partial \vc{B}}{\partial t} \\
            \rot \vc{H} &= \frac{4\pi}{c} \vc{j} + \frac{1}{c} \frac{\partial \vc{D}}{\partial t}.
    \end{aligned}\right.
    \hspace{0.5cm} \Rightarrow \hspace{0.5cm}
    \rot \rot \vc{E} = - \frac{1}{c} \frac{\partial }{\partial t} \rot \vc{B}
    \hspace{0.5cm} \Rightarrow \hspace{0.5cm}
    \nabla^2 \vc{E} - \frac{\varepsilon \mu}{c^2} \frac{\partial^2 \vc{E}}{\partial t^2} = 0
\end{equation*}

\textbf{–ü—É—Ç—å –Ω–∞–ø—Ä—è–º–∏–∫}.
–ü–æ–¥—Å—Ç–∞–≤–ª—è—è $E = a(\vc{r}) e^{i(\omega t - \smallvc{k}_0 \Phi(\smallvc{r}))}$ –≤ —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –ì–µ–ª—å–º–≥–æ–ª—å—Ü–∞ –º–æ–∂–µ–º –ø–æ–ª—É—á–∏—Ç—å —Å–ª–µ–¥—É—é—â—É—é —Å–∏—Å—Ç–µ–º—É:
\begin{align*}
    (\grad \Phi)^2 = n^2 + \frac{\Delta a}{k_0^2 a} \\
    a \Delta \Phi + 2 (\grad a) (\grad \Phi) = 0
\end{align*}
–í –∫–æ—Ç–æ—Ä–æ–º —Ö–æ—Ç–∏–º —É–≤–∏–¥–µ—Ç—å
\begin{equation*}
    \bigg| \frac{\Delta a}{k_0^2 a} \bigg| = \bigg| \frac{\Delta a}{a} \bigg| \frac{\lambda^2}{4 \pi^2} \ll n^2 
    \hspace{2.5 mm} \Leftrightarrow \hspace{2.5 mm} 
    \left\{\begin{aligned}
        \lambda |\partial_x a| \ll a, \\
        \lambda |\partial_{xx}^2 a| \ll |\partial_x a|,
    \end{aligned}\right.
\hspace{2.5 mm} \Rightarrow \hspace{2.5 mm}
(\grad \Phi)^2 = n^2,
\hspace{2.5 mm} \Leftrightarrow \hspace{2.5 mm} 
\grad \Phi = n \vc{s}.
\end{equation*}
–ò–∑ –≤—Ç–æ—Ä–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è –º–æ–∂–µ–º –ø–æ–ª—É—á–∏—Ç—å –Ω–µ–∑–∞–≤–∏—Å–∏–º–æ—Å—Ç—å –ø–æ–≤–µ–¥–µ–Ω–∏—è –ª—É—á–µ–π –¥—Ä—É–≥ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –¥—Ä—É–≥–∞. 


\textbf{–í —Ç–µ—Ä–º–∏–Ω–∞—Ö 4-–≤–µ–∫—Ç–æ—Ä–æ–≤}. –î–∞–≤–∞–π—Ç–µ —Å–∫–∞–∂–µ–º, —á—Ç–æ –Ω–∞—à–µ —Å–∫–∞–ª—è—Ä–Ω–æ–µ –ø–æ–ª–µ —ç—Ç–æ
$$f = a e^{i(\smallvc{k} \smallvc{r} - \omega t + \alpha)} = a e^{i \Phi}$$, –≥–¥–µ $\Phi$ -- –≠–π–∫–æ–Ω–∞–ª. –¢–æ–≥–¥–∞ $\grad \Phi = \vc{k}$, $\partial_t \Phi = - \omega$, –∏ –≤–ø–æ–ª–Ω–µ –µ—Å—Ç–µ—Å—Ç–≤–µ–Ω–Ω–æ –≥–æ–≤–æ—Ä–∏—Ç—å –ø—Ä–æ 4-–≤–µ–∫—Ç–æ—Ä $k^i = \left(\frac{\omega}{c},\, \vc{k}\right)\T$, —Ç–æ–≥–¥–∞ $k_i x^i = \omega t - \vc{k} \cdot \vc{r}$ –∏ 
\begin{equation*}
    k_i k^i = \frac{\omega^2}{c^2} - k^2 = 0,
    \hspace{0.5cm} \Rightarrow \hspace{0.5cm}
    \partial_i \Phi \partial_i \Phi = 0.
\end{equation*}
–¢–æ–≥–¥–∞ –ø—É—Å—Ç—å $\Phi = - \omega t + \Phi_0 (x, y, z)$, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ $\partial^i \Phi = \left(\frac{\omega}{c},\, \frac{\partial \Phi_0}{\partial \smallvc{r}} \right)$, —Ç–∞–∫ –ø—Ä–∏—Ö–æ–¥–∏–º –∫
\begin{equation*}
    \left(\grad \Phi_0\right)^2 = \frac{\omega^2}{c^2} =  k_0^2,
    \hspace{0.5cm} \overset{c \to c/n}{\Rightarrow}  \hspace{0.5cm}
    \left(\grad \Phi_0 / k_0\right)^2 = n^2.
\end{equation*}


\textbf{–ü—Ä–∏–Ω—Ü–∏–ø –§–µ—Ä–º–∞}. –†–∞—Å—Å–º–æ—Ç—Ä–∏–º –∏–Ω—Ç–µ–≥—Ä–∞–ª –ø–æ –∑–∞–º–∫–Ω–æ—Ç–æ–º—É –∫–æ–Ω—Ç–æ—Ä—É
\begin{equation*}
    \oint \grad \Phi \cdot \d \vc{l} = \oint n (\vc{s} \cdot \d \vc{l}) = 0
\end{equation*}
–ù–æ —ç—Ç–æ —Å–æ–æ—Ç–≤–µ—Ç—Å–≤—É–µ—Ç —ç–∫—Å—Ç—Ä–µ–º–∞–ª—å–Ω–æ—Å—Ç–∏ $\int n \d l$, —Ç–æ –µ—Å—Ç—å
\begin{equation*}
    \delta \int n \d l  = 0,
\end{equation*}
–¥–ª—è –ø—Ä—è–º–æ–≥–æ –ø—É—Ç–∏. 

–ß—É—Ç—å –±–æ–ª–µ–µ —è–≤–Ω–æ –º–æ–∂–µ–º –≤—ã–ø–∏—Å–∞—Ç—å $\grad \Phi = n \vc{s}$, —Ç–æ–≥–¥–∞ $d_l \Phi = n$, –∏ $\vc{s} = d_l \vc{r}$, —Ç–æ–≥–¥–∞
\begin{equation*}
    \frac{d }{d l} \left(n \frac{d \vc{r}}{d l} \right) = \frac{d }{d l}  \grad \Phi = \grad n.
\end{equation*}

\textbf{–ö–æ–æ—Ä–¥–∏–Ω–∞—Ç–Ω–∞—è –∑–∞–ø–∏—Å—å}. –ó–∞–º–µ—Ç–∏–º, —á—Ç–æ $\d l = \sqrt{1 + (y'_x)^2} \d x$, —Ç–æ–≥–¥–∞
\begin{equation*}
    \delta \int n \sqrt{1 + (y'_x)^2} \d x = 0,
\end{equation*}
—á—Ç–æ —Å–æ–æ—Ç–≤–µ—Ç—Å–≤—É–µ—Ç —É—Ä–∞–≤–Ω–µ–Ω–∏—é –≠–π–ª–µ—Ä–∞-–õ–∞–≥—Ä–∞–Ω–∂–∞:
\begin{equation*}
    \frac{d }{d x}  \frac{\partial L}{\partial y'_x} - \frac{\partial L}{\partial y} = 0,
    \hspace{5 mm}   
    L = n \sqrt{1 + (y'_x)^2}, \hspace{5 mm} 
    n \equiv n(x, y).
\end{equation*}


\textbf{–û–ø—Ç–∏—á–µ—Å–∫–∏–π —ç–ª–µ–º–µ–Ω—Ç —Å –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–º $n$}. –†–∞—Å—Å–º–æ—Ç—Ä–∏–º, –Ω–∞–ø—Ä–∏–º–µ—Ä $n^2 = n_0^2 (1-\alpha^2 y^2)$, —Ä–∞—Å—Å–º–æ—Ç—Ä–∏–º –ø–∞—Ä–∞–∫—Å–∏–∞–ª—å–Ω–æ–µ –ø—Ä–∏–±–ª–∏–∂–µ–Ω–∏–µ –∏ —É–≤–∏–¥–∏–º
\begin{equation*}
    y''_{xx} = \frac{1}{n} \frac{d n}{d y},
    \hspace{0.5cm} \Rightarrow \hspace{0.5cm}
    y''_{xx} = - \alpha^2 y,
\end{equation*}
—á—Ç–æ –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç, —á—Ç–æ –º—ã —Å–Ω–æ–≤–∞ —Å–≤–µ–ª–∏ –≤—Å—ë –∫ –≥–∞—Ä–º–æ–Ω–∏—á–µ—Å–∫–æ–º—É –æ—Å—Ü–∏–ª–ª—è—Ç–æ—Ä—É.

–¢–∞–∫ —Ñ–æ–∫—É—Å–Ω–æ–µ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –¥–ª—è —Å—Ñ–µ—Ä–∏—á–µ—Å–∫–æ–≥–æ –∑–µ—Ä–∫–∞–ª–∞ $R/2$, —á—Ç–æ –ª–æ–≥–∏—á–Ω–æ. –í —Å–ª—É—á–∞–µ –∂–µ, –µ—Å–ª–∏ –º—ã –∑–∞—Ö–æ—Ç–∏–º —Å–ª–µ–¥–∏—Ç—å –∑–∞ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è–º–∏ –æ—Å–µ–π, —Ç–æ –º–æ–∂–Ω–æ –≤–µ—Ä–Ω—É—Ç—å—Å—è –∫ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–º $\{z, \theta\}$.

–∏, –Ω–∞–∫–æ–Ω–µ—Ü,
\begin{equation*}

     y_m = 0,
\end{equation*}
–∫–æ—Ç–æ—Ä–æ–µ —Ä–µ—à–∞–µ—Ç—Å—è —Ç–∞–∫–∂–µ, –∫–∞–∫ –∏ –¥–∏—Ñ—Ñ—É—Ä, –ø–æ–¥—Å—Ç–∞–Ω–æ–≤–∫–æ–π $y_m = y_0 h^m$, —Ç–æ–≥–¥–∞
\begin{equation*}
    y_m = y_0 h^m, \hspace{0.5cm} \Rightarrow \hspace{0.5cm}
    h^2 - (A+D) h + 1 = 0,
\end{equation*}
—Å—á–∏—Ç–∞—è $\tr M = A+D \overset{\mathrm{def}}{=} 2 b$, –Ω–∞—Ö–æ–¥–∏–º, —á—Ç–æ
\begin{equation*}
    h^2 - 2 b h  +1 = 0, 
    \hspace{0.5cm} \Rightarrow \hspace{0.5cm}
    h_{1, 2} = b \pm \sqrt{b^2-1}
\end{equation*}
—á—Ç–æ –ø—Ä–∏–≤–æ–¥–∏—Ç –Ω–∞—Å –∫ –Ω–µ–∫–æ—Ç–æ—Ä–æ–º—É –∫ —Å–ª–µ–¥—É—é—â–µ–π –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏

$|b|>1$ -- –Ω–µ—É—Å—Ç–æ–π—á–∏–≤—ã–π —Ä–µ–∂–∏–º

$|b|=1$ -- –≥—Ä–∞–Ω–∏—Ü–∞

$|b|<1$ -- —É—Å—Ç–æ–π—á–∏–≤–æ—Å—Ç—å

\noindent
–†–∞—Å—Å–º–æ—Ç—Ä–∏–º $b < 1$ –∏ –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞ –ø–æ–ª–æ–∂–∏–º $b = \cos \varphi$, —Ç–æ–≥–¥–∞
\begin{equation*}
    h_{1, 2} = \cos \varphi \pm i \sin \varphi = e^{i\varphi},
    \hspace{0.5cm} \Rightarrow \hspace{0.5cm}
    y_m = \alpha_1 e^{im \varphi} + \alpha_2 e^{-im \varphi} = 
    \underline{y_{\textnormal{max}}} \sin (m \varphi + \underline{\varphi_0}),
\end{equation*}
–≥–¥–µ –ø–æ–¥—á–µ—Ä–∫–Ω—É—Ç—ã–µ –ø–∞—Ä–∞–µ–º—Ç—Ä—ã –æ–ø—Ä–µ–¥–µ–ª—è—é—Ç—Å—è –Ω–∞—á–∞–ª—å–Ω—ã–º–∏ —É—Å–ª–æ–≤–∏—è–º–∏. 


–°—Ç–æ–∏—Ç –∑–∞–º–µ—Ç–∏—Ç—å, —á—Ç–æ —Ö–æ—Ç–µ–ª–æ—Å—å –±—ã $\theta_m$ –ø–µ—Ä–∏–æ–¥–∏—á–µ—Å–∫–æ–π. 




\subsubsection*{–ü—Ä–∏–º–µ—Ä (–û–ø—Ç–∏—á–µ—Å–∫–∏–π —Ä–µ–∑–æ–Ω–∞—Ç–æ—Ä)}

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏–µ —Å–≤–µ—Ç–∞ –Ω–∞ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ $L$ –º–µ–∂–¥—É –¥–≤—É–º—è –∑–µ—Ä–∫–∞–ª–∞–º–∏ –∫—Ä–∏–≤–∏–∑—ã $R_1,\, R_2$, —Ç–æ–≥–¥–∞ $b$ 
\begin{equation*}
    b = 2 \bigg(\underbrace{1+\frac{L}{R_1}}_{g_1}\bigg)\bigg(\underbrace{1+\frac{L}{R_2}}_{g_2}\bigg) - 1,
\end{equation*}
–∏ —Ä–∞—Å—Å–º–æ—Ç—Ä–∏–º $0 \leq g_1 g_2 \leq 1$.


–ü–µ—Ä–≤—ã–π —Å–ª—É—á–∞–π (1) –ø–ª–æ—Å–∫–æ–≥–æ —Ä–µ–∑–æ–Ω–∞—Ç–æ—Ä–∞ –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –Ω–∞ –≥—Ä–∞–Ω—Ü–µ. –î—Ä—É–≥–æ–π —Å–ª—É—á–∞–π (2) —ç—Ç–æ —Å–∏–º–º–µ—Ç—Ä–∏—á–Ω—ã–π –∫–æ—Ñ–æ–∫–∞–Ω—ã–π —Ä–µ–∑–æ–Ω–∞—Ç–æ—Ä, —Ç–æ–≥–¥–∞ $R_1 = -L$, –∏ $R_2 = - L$.  –í–æ–∑–º–æ–∂–µ–Ω —Å–∏–º–º–µ—Ç—Ä–∏—á–Ω—ã–π –∫–æ–Ω—Ü–µ–Ω—Ç—Ä–∏—á–µ—Å–∫–∏–π $R_1 = R_2 = -L/2$. 


–ü—É—Å—Ç—å –µ—Å—Ç—å –Ω–µ–∫–æ—Ç–æ—Ä–∞—è –∞–∫—Ç–∏–≤–Ω–∞—è —Å—Ä–µ–¥–∞, –ø—Ä–æ—Ü–µ—Å—Å –Ω–∞–∫–∞—á–∫–∏.

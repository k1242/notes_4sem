		\begin{aligned}
			&R_\perp = \frac{sin^2(\theta_2 - \theta_0)}{\sin^2(\theta_2 + \theta_0)},\\
			\hspace{1 cm}
			&R_\parallel = \frac{\tg^2(\theta_2 - \theta_0)}{\tg^2(\theta_2 + \theta_0)}.	
		\end{aligned}
	\end{equation}

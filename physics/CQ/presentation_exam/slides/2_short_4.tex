\begin{equation}
	\frac{d S}{d t} = \Gamma_{MD} G_0 v_g S - \frac{S}{\tau_p}
    \hspace{0.3 cm}
    \Rightarrow
    \hspace{0.3 cm}
    \frac{d S}{d t} = v_g g S(t) - \gamma_p S. 
\end{equation}
\begin{equation}
	\frac{d N}{d t} = \frac{J}{e} - \frac{N}{\tau_n} - \Gamma_{MD} G_0 v_g S 
    \hspace{0.3 cm}
    \Rightarrow
    \hspace{0.3 cm}
    \frac{d N}{d t}
    = \frac{J}{e d} - \gamma_s - g S.
\end{equation}

\begin{equation}
